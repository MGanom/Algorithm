<h1>문제</h1>

2020 KAKAO BLIND RECRUITMENT - 괄호 변환 <br />
https://programmers.co.kr/learn/courses/30/lessons/60058


<h1>풀이</h1>
<h2>처음에 한 풀이(풀이 중)</h2>

```jsx
function solution(p) {
  if (p === '') {
    return '';
  }

  let pair = '';
  let check = '';
  let count1 = 0;
  let count2 = 0;
  let valid = 'o';
  for (i = 0; i < p.length; i++) {
    check += p[i];
    if (p[i] === '(') {
      count1++;
    } else {
      count2++;
    }
    if (count1 < count2) {
      valid = 'x';
    }
    if (count1 === count2) {
      let u = check;
      let v = p.slice(i + 1);
      if (valid === 'o') {
        pair = pair + u;
        pair = pair + solution(v);
        return pair;
      } else {
        pair = pair + '(';
        pair = pair + solution(v);
        pair = pair + ')';
        for (let j = 1; i > j; j++) {
          if (p[j] === ')') {
            pair = pair + '(';
          } else {
            pair = pair + ')';
          }
        }
        return pair;
      }
    }
  }
}
```

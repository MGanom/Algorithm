```jsx
function solution(nodeinfo) {
  let answer = [];

  let nodelist = {};
  for (let i = 0; i < nodeinfo.length; i++) {
    nodelist[i + 1] = nodeinfo[i];
  }

  console.log(Object.values(nodelist));
  Object.values(nodelist).sort((a, b) => {
    if (a[1] > b[1]) {
      return -1;
    }
  });

  console.log(nodelist);

  return answer;
}
```

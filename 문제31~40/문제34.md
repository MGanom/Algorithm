```jsx
function solution(key, lock) {
  function rotate(arr) {
    let rotated = [];
    for (let i = 0; i < arr[0].length; i++) {
      let key = [];
      for (let j = arr.length - 1; j > -1; j--) {
        key.push(arr[j][i]);
      }
      rotated.push(key);
    }
    return rotated;
  }
    
}
```

```jsx
function solution(stones, k) {
  let answer = 0;
  let jump = 0;
  let hole = [];
  function cross(arr) {
    if (jump !== k) {
      if (!arr.includes(0)) {
        answer++;
      }
      for (let i = 0; i < arr.length; i++) {
        if (arr[i] !== 0) {
          arr[i] = arr[i] - 1;
        }
      }
      if (arr.includes(0)) {
        for (let i = 0; i < arr.length; i++) {
          if (arr[i] === 0 && !hole.includes(i)) {
            hole.push(i);
          }
        }
        if (jump === 0) {
          jump = 1;
        }
        for (let i = 0; i < hole.length - 1; i++) {
          if (hole[i] + 1 === hole[i + 1]) {
            jump++;
          }
        }
      }
      // cross(arr);
    }
  }
  cross(stones);
  console.log(hole);
  console.log(jump);
  console.log(stones);
  console.log(answer);
}
```
